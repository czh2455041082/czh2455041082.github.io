show character set;

2．查看该’gb2312’字符集对应的校对原则

show collation like 'gb2312';

3．查看该数据库系统支持的存储引擎的类型

show engines;

4．创建student数据库和test数据库，其中，Student数据库指定字符集位gb2312，校对原则为gb2312_chinese_ci

create database student default character set gb2312 default collate gb2312_chinese_ci;
create database if not exists test;

5．查看数据库系统中已经存在的数据库，确保student和test数据库已经存在

show databases;

6． 显示test数据库的结构信息

show create database test;

7．修改test数据库的指定字符集位gb2312，校对原则为gb2312_chinese_ci

alter database test default character set gb2312 default collate gb2312_chinese_ci;

8．再次显示test数据库的结构信息

show create database test;

9． 删除student和test数据库

drop database student;
drop database test;
